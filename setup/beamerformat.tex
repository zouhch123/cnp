% 增加首页及目录
\AtBeginLecture{
	\begin{frame}
		\titlepage
		\addtocounter{framenumber}{-2}
		\thispagestyle{empty}
	\end{frame}

	\begin{frame}{本讲内容}
		\Large
		\setbeamertemplate{section in toc}[sections numbered]
		\tableofcontents[hideallsubsections]
		\thispagestyle{empty}
	\end{frame}

}
